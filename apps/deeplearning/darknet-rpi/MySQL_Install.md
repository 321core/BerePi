# 구축

# 명령어 
  &nbsp; mysql 버전 상태 확인 : mysql --version <br>
  &nbsp; mysql port 설정 확인 : cat /etc/mysql/my.cnf | grep port <br>
  &nbsp; mysql 서비스 상태 확인, 시작, 정지 : service mysql status, start, stop <br>
  &nbsp; Linux port 상태 확인 : nmap localhost <br>
